    def gwo_experiment(id, uacct, sections = [], options = {}, &block)
      options = {
        :conditions => true,
        :ga_tracking => false,
        :ga_base_url => nil
      }.update(options)

      src  = gwo_start(id, sections, options)
      src += capture(&block) 
      src += gwo_end(id, uacct, options)
      src
    end
